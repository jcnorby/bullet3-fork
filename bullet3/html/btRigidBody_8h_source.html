391  SIMD_FORCE_INLINE btScalar computeImpulseDenominator(const btVector3& pos, const btVector3& normal) const
392  {
393  btVector3 r0 = pos - getCenterOfMassPosition();
394 
395  btVector3 c0 = (r0).cross(normal);
396 
397  btVector3 vec = (c0 * getInvInertiaTensorWorld()).cross(r0);
398 
399  return m_inverseMass + normal.dot(vec);
400  }

408  SIMD_FORCE_INLINE void updateDeactivation(btScalar timeStep)
409  {
410  if ((getActivationState() == ISLAND_SLEEPING) || (getActivationState() == DISABLE_DEACTIVATION))
411  return;
412 
413  if ((getLinearVelocity().length2() < m_linearSleepingThreshold * m_linearSleepingThreshold) &&
414  (getAngularVelocity().length2() < m_angularSleepingThreshold * m_angularSleepingThreshold))
415  {
416  m_deactivationTime += timeStep;
417  }
418  else
419  {
420  m_deactivationTime = btScalar(0.);
421  setActivationState(0);
422  }
423  }
424 
425  SIMD_FORCE_INLINE bool wantsSleeping()
426  {
427  if (getActivationState() == DISABLE_DEACTIVATION)
428  return false;
429 
430  //disable deactivation
431  if (gDisableDeactivation || (gDeactivationTime == btScalar(0.)))
432  return false;
433 
434  if ((getActivationState() == ISLAND_SLEEPING) || (getActivationState() == WANTS_DEACTIVATION))
435  return true;
436 
437  if (m_deactivationTime > gDeactivationTime)
438  {
439  return true;
440  }
441  return false;
442  }
