36 void btMultiBodyDynamicsWorld::calculateSimulationIslands()
37 {
38  BT_PROFILE("calculateSimulationIslands");
39 
40  getSimulationIslandManager()->updateActivationState(getCollisionWorld(), getCollisionWorld()->getDispatcher());
41 
42  {
43  //merge islands based on speculative contact manifolds too
44  for (int i = 0; i < this->m_predictiveManifolds.size(); i++)
45  {
46  btPersistentManifold* manifold = m_predictiveManifolds[i];
47 
48  const btCollisionObject* colObj0 = manifold->getBody0();
49  const btCollisionObject* colObj1 = manifold->getBody1();
50 
51  if (((colObj0) && (!(colObj0)->isStaticOrKinematicObject())) &&
52  ((colObj1) && (!(colObj1)->isStaticOrKinematicObject())))
53  {
54  getSimulationIslandManager()->getUnionFind().unite((colObj0)->getIslandTag(), (colObj1)->getIslandTag());
55  }
56  }
57  }
58 
59  {
60  int i;
61  int numConstraints = int(m_constraints.size());
62  for (i = 0; i < numConstraints; i++)
63  {
64  btTypedConstraint* constraint = m_constraints[i];
65  if (constraint->isEnabled())
66  {
67  const btRigidBody* colObj0 = &constraint->getRigidBodyA();
68  const btRigidBody* colObj1 = &constraint->getRigidBodyB();
69 
70  if (((colObj0) && (!(colObj0)->isStaticOrKinematicObject())) &&
71  ((colObj1) && (!(colObj1)->isStaticOrKinematicObject())))
72  {
73  getSimulationIslandManager()->getUnionFind().unite((colObj0)->getIslandTag(), (colObj1)->getIslandTag());
74  }
75  }
76  }
77  }
78 
79  //merge islands linked by Featherstone link colliders
80  for (int i = 0; i < m_multiBodies.size(); i++)
81  {
82  btMultiBody* body = m_multiBodies[i];
83  {
84  btMultiBodyLinkCollider* prev = body->getBaseCollider();
85 
86  for (int b = 0; b < body->getNumLinks(); b++)
87  {
88  btMultiBodyLinkCollider* cur = body->getLink(b).m_collider;
89 
90  if (((cur) && (!(cur)->isStaticOrKinematicObject())) &&
91  ((prev) && (!(prev)->isStaticOrKinematicObject())))
92  {
93  int tagPrev = prev->getIslandTag();
94  int tagCur = cur->getIslandTag();
95  getSimulationIslandManager()->getUnionFind().unite(tagPrev, tagCur);
96  }
97  if (cur && !cur->isStaticOrKinematicObject())
98  prev = cur;
99  }
100  }
101  }
102 
103  //merge islands linked by multibody constraints
104  {
105  for (int i = 0; i < this->m_multiBodyConstraints.size(); i++)
106  {
107  btMultiBodyConstraint* c = m_multiBodyConstraints[i];
108  int tagA = c->getIslandIdA();
109  int tagB = c->getIslandIdB();
110  if (tagA >= 0 && tagB >= 0)
111  getSimulationIslandManager()->getUnionFind().unite(tagA, tagB);
112  }
113  }
114 
115  //Store the island id in each body
116  getSimulationIslandManager()->storeIslandActivationState(getCollisionWorld());
117 }

271  virtual void processIsland(btCollisionObject** bodies, int numBodies, btPersistentManifold** manifolds, int numManifolds, int islandId)
272  {
273  if (islandId < 0)
274  {
276  m_solver->solveMultiBodyGroup(bodies, numBodies, manifolds, numManifolds, m_sortedConstraints, m_numConstraints, &m_multiBodySortedConstraints[0], m_numConstraints, *m_solverInfo, m_debugDrawer, m_dispatcher);
277  if (m_solverInfo->m_reportSolverAnalytics&1)
278  {
279  m_solver->m_analyticsData.m_islandId = islandId;
280  m_islandAnalyticsData.push_back(m_solver->m_analyticsData);
281  }
282  }
283  else
284  {
285  //also add all non-contact constraints/joints for this island
286  btTypedConstraint** startConstraint = 0;
287  btMultiBodyConstraint** startMultiBodyConstraint = 0;
288 
289  int numCurConstraints = 0;
290  int numCurMultiBodyConstraints = 0;
291 
292  int i;
293 
294  //find the first constraint for this island
295 
296  for (i = 0; i < m_numConstraints; i++)
297  {
298  if (btGetConstraintIslandId2(m_sortedConstraints[i]) == islandId)
299  {
300  startConstraint = &m_sortedConstraints[i];
301  break;
302  }
303  }
304  //count the number of constraints in this island
305  for (; i < m_numConstraints; i++)
306  {
307  if (btGetConstraintIslandId2(m_sortedConstraints[i]) == islandId)
308  {
309  numCurConstraints++;
310  }
311  }
312 
313  for (i = 0; i < m_numMultiBodyConstraints; i++)
314  {
315  if (btGetMultiBodyConstraintIslandId(m_multiBodySortedConstraints[i]) == islandId)
316  {
317  startMultiBodyConstraint = &m_multiBodySortedConstraints[i];
318  break;
319  }
320  }
321  //count the number of multi body constraints in this island
322  for (; i < m_numMultiBodyConstraints; i++)
323  {
324  if (btGetMultiBodyConstraintIslandId(m_multiBodySortedConstraints[i]) == islandId)
325  {
326  numCurMultiBodyConstraints++;
327  }
328  }
329 
330  //if (m_solverInfo->m_minimumSolverBatchSize<=1)
331  //{
332  // m_solver->solveGroup( bodies,numBodies,manifolds, numManifolds,startConstraint,numCurConstraints,*m_solverInfo,m_debugDrawer,m_dispatcher);
333  //} else
334  {
335  for (i = 0; i < numBodies; i++)
336  m_bodies.push_back(bodies[i]);
337  for (i = 0; i < numManifolds; i++)
338  m_manifolds.push_back(manifolds[i]);
339  for (i = 0; i < numCurConstraints; i++)
340  m_constraints.push_back(startConstraint[i]);
341 
342  for (i = 0; i < numCurMultiBodyConstraints; i++)
343  m_multiBodyConstraints.push_back(startMultiBodyConstraint[i]);
344 
345  if ((m_multiBodyConstraints.size() + m_constraints.size() + m_manifolds.size()) > m_solverInfo->m_minimumSolverBatchSize)
346  {
347  processConstraints(islandId);
348  }
349  else
350  {
351  //printf("deferred\n");
352  }
353  }
354  }
355  }
356  void processConstraints(int islandId=-1)
357  {
358  btCollisionObject** bodies = m_bodies.size() ? &m_bodies[0] : 0;
359  btPersistentManifold** manifold = m_manifolds.size() ? &m_manifolds[0] : 0;
360  btTypedConstraint** constraints = m_constraints.size() ? &m_constraints[0] : 0;
361  btMultiBodyConstraint** multiBodyConstraints = m_multiBodyConstraints.size() ? &m_multiBodyConstraints[0] : 0;
362 
363  //printf("mb contacts = %d, mb constraints = %d\n", mbContacts, m_multiBodyConstraints.size());
364 
365  m_solver->solveMultiBodyGroup(bodies, m_bodies.size(), manifold, m_manifolds.size(), constraints, m_constraints.size(), multiBodyConstraints, m_multiBodyConstraints.size(), *m_solverInfo, m_debugDrawer, m_dispatcher);
366  if (m_bodies.size() && (m_solverInfo->m_reportSolverAnalytics&1))
367  {
368  m_solver->m_analyticsData.m_islandId = islandId;
369  m_islandAnalyticsData.push_back(m_solver->m_analyticsData);
370  }
371  m_bodies.resize(0);
372  m_manifolds.resize(0);
373  m_constraints.resize(0);
374  m_multiBodyConstraints.resize(0);
375  }
