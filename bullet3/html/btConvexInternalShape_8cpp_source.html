50 btVector3 btConvexInternalShape::localGetSupportingVertex(const btVector3& vec) const
51 {
52 #ifndef __SPU__
53 
54  btVector3 supVertex = localGetSupportingVertexWithoutMargin(vec);
55 
56  if (getMargin() != btScalar(0.))
