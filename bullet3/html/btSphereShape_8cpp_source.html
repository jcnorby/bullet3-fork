37 btVector3 btSphereShape::localGetSupportingVertex(const btVector3& vec) const
38 {
39  btVector3 supVertex;
40  supVertex = localGetSupportingVertexWithoutMargin(vec);
41 
42  btVector3 vecnorm = vec;
43  if (vecnorm.length2() < (SIMD_EPSILON * SIMD_EPSILON))
44  {
