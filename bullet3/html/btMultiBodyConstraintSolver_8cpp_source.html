26 btScalar btMultiBodyConstraintSolver::solveSingleIteration(int iteration, btCollisionObject** bodies, int numBodies, btPersistentManifold** manifoldPtr, int numManifolds, btTypedConstraint** constraints, int numConstraints, const btContactSolverInfo& infoGlobal, btIDebugDraw* debugDrawer)
27 {
28  btScalar leastSquaredResidual = btSequentialImpulseConstraintSolver::solveSingleIteration(iteration, bodies, numBodies, manifoldPtr, numManifolds, constraints, numConstraints, infoGlobal, debugDrawer);
29 
30  //solve featherstone non-contact constraints
31 
32  //printf("m_multiBodyNonContactConstraints = %d\n",m_multiBodyNonContactConstraints.size());
33 
34  for (int j = 0; j < m_multiBodyNonContactConstraints.size(); j++)
35  {
36  int index = iteration & 1 ? j : m_multiBodyNonContactConstraints.size() - 1 - j;
37 
38  btMultiBodySolverConstraint& constraint = m_multiBodyNonContactConstraints[index];
39 
40  btScalar residual = resolveSingleConstraintRowGeneric(constraint);
41  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
42 
43  if (constraint.m_multiBodyA)
44  constraint.m_multiBodyA->setPosUpdated(false);
45  if (constraint.m_multiBodyB)
46  constraint.m_multiBodyB->setPosUpdated(false);
47  }
48 
49  //solve featherstone normal contact
50  for (int j0 = 0; j0 < m_multiBodyNormalContactConstraints.size(); j0++)
51  {
52  int index = j0; //iteration&1? j0 : m_multiBodyNormalContactConstraints.size()-1-j0;
53 
54  btMultiBodySolverConstraint& constraint = m_multiBodyNormalContactConstraints[index];
55  btScalar residual = 0.f;
56 
57  if (iteration < infoGlobal.m_numIterations)
58  {
59  residual = resolveSingleConstraintRowGeneric(constraint);
60  }
61 
62  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
63 
64  if (constraint.m_multiBodyA)
65  constraint.m_multiBodyA->setPosUpdated(false);
66  if (constraint.m_multiBodyB)
67  constraint.m_multiBodyB->setPosUpdated(false);
68  }
69 
70  //solve featherstone frictional contact
71  if (infoGlobal.m_solverMode & SOLVER_USE_2_FRICTION_DIRECTIONS && ((infoGlobal.m_solverMode & SOLVER_DISABLE_IMPLICIT_CONE_FRICTION) == 0))
72  {
73  for (int j1 = 0; j1 < this->m_multiBodySpinningFrictionContactConstraints.size(); j1++)
74  {
75  if (iteration < infoGlobal.m_numIterations)
76  {
77  int index = j1;
78 
79  btMultiBodySolverConstraint& frictionConstraint = m_multiBodySpinningFrictionContactConstraints[index];
80  btScalar totalImpulse = m_multiBodyNormalContactConstraints[frictionConstraint.m_frictionIndex].m_appliedImpulse;
81  //adjust friction limits here
82  if (totalImpulse > btScalar(0))
83  {
84  frictionConstraint.m_lowerLimit = -(frictionConstraint.m_friction * totalImpulse);
85  frictionConstraint.m_upperLimit = frictionConstraint.m_friction * totalImpulse;
86  btScalar residual = resolveSingleConstraintRowGeneric(frictionConstraint);
87  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
88 
89  if (frictionConstraint.m_multiBodyA)
90  frictionConstraint.m_multiBodyA->setPosUpdated(false);
91  if (frictionConstraint.m_multiBodyB)
92  frictionConstraint.m_multiBodyB->setPosUpdated(false);
93  }
94  }
95  }
96 
97  for (int j1 = 0; j1 < this->m_multiBodyTorsionalFrictionContactConstraints.size(); j1++)
98  {
99  if (iteration < infoGlobal.m_numIterations)
100  {
101  int index = j1; //iteration&1? j1 : m_multiBodyTorsionalFrictionContactConstraints.size()-1-j1;
102 
103  btMultiBodySolverConstraint& frictionConstraint = m_multiBodyTorsionalFrictionContactConstraints[index];
104  btScalar totalImpulse = m_multiBodyNormalContactConstraints[frictionConstraint.m_frictionIndex].m_appliedImpulse;
105  j1++;
106  int index2 = j1;
107  btMultiBodySolverConstraint& frictionConstraintB = m_multiBodyTorsionalFrictionContactConstraints[index2];
108  //adjust friction limits here
109  if (totalImpulse > btScalar(0) && frictionConstraint.m_frictionIndex == frictionConstraintB.m_frictionIndex)
110  {
111  frictionConstraint.m_lowerLimit = -(frictionConstraint.m_friction * totalImpulse);
112  frictionConstraint.m_upperLimit = frictionConstraint.m_friction * totalImpulse;
113  frictionConstraintB.m_lowerLimit = -(frictionConstraintB.m_friction * totalImpulse);
114  frictionConstraintB.m_upperLimit = frictionConstraintB.m_friction * totalImpulse;
115 
116  btScalar residual = resolveConeFrictionConstraintRows(frictionConstraint, frictionConstraintB);
117  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
118 
119  if (frictionConstraint.m_multiBodyA)
120  frictionConstraint.m_multiBodyA->setPosUpdated(false);
121  if (frictionConstraint.m_multiBodyB)
122  frictionConstraint.m_multiBodyB->setPosUpdated(false);
123 
124  if (frictionConstraintB.m_multiBodyA)
125  frictionConstraintB.m_multiBodyA->setPosUpdated(false);
126  if (frictionConstraintB.m_multiBodyB)
127  frictionConstraintB.m_multiBodyB->setPosUpdated(false);
128  }
129  }
130  }
131 
132  for (int j1 = 0; j1 < this->m_multiBodyFrictionContactConstraints.size(); j1++)
133  {
134  if (iteration < infoGlobal.m_numIterations)
135  {
136  int index = j1; //iteration&1? j1 : m_multiBodyFrictionContactConstraints.size()-1-j1;
137  btMultiBodySolverConstraint& frictionConstraint = m_multiBodyFrictionContactConstraints[index];
138 
139  btScalar totalImpulse = m_multiBodyNormalContactConstraints[frictionConstraint.m_frictionIndex].m_appliedImpulse;
140  j1++;
141  int index2 = j1; //iteration&1? j1 : m_multiBodyFrictionContactConstraints.size()-1-j1;
142  btMultiBodySolverConstraint& frictionConstraintB = m_multiBodyFrictionContactConstraints[index2];
143  btAssert(frictionConstraint.m_frictionIndex == frictionConstraintB.m_frictionIndex);
144 
145  if (frictionConstraint.m_frictionIndex == frictionConstraintB.m_frictionIndex)
146  {
147  frictionConstraint.m_lowerLimit = -(frictionConstraint.m_friction * totalImpulse);
148  frictionConstraint.m_upperLimit = frictionConstraint.m_friction * totalImpulse;
149  frictionConstraintB.m_lowerLimit = -(frictionConstraintB.m_friction * totalImpulse);
150  frictionConstraintB.m_upperLimit = frictionConstraintB.m_friction * totalImpulse;
151  btScalar residual = resolveConeFrictionConstraintRows(frictionConstraint, frictionConstraintB);
152  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
153 
154  if (frictionConstraintB.m_multiBodyA)
155  frictionConstraintB.m_multiBodyA->setPosUpdated(false);
156  if (frictionConstraintB.m_multiBodyB)
157  frictionConstraintB.m_multiBodyB->setPosUpdated(false);
158 
159  if (frictionConstraint.m_multiBodyA)
160  frictionConstraint.m_multiBodyA->setPosUpdated(false);
161  if (frictionConstraint.m_multiBodyB)
162  frictionConstraint.m_multiBodyB->setPosUpdated(false);
163  }
164  }
165  }
166  }
167  else
168  {
169  for (int j1 = 0; j1 < this->m_multiBodyFrictionContactConstraints.size(); j1++)
170  {
171  if (iteration < infoGlobal.m_numIterations)
172  {
173  int index = j1; //iteration&1? j1 : m_multiBodyFrictionContactConstraints.size()-1-j1;
174 
175  btMultiBodySolverConstraint& frictionConstraint = m_multiBodyFrictionContactConstraints[index];
176  btScalar totalImpulse = m_multiBodyNormalContactConstraints[frictionConstraint.m_frictionIndex].m_appliedImpulse;
177  //adjust friction limits here
178  if (totalImpulse > btScalar(0))
179  {
180  frictionConstraint.m_lowerLimit = -(frictionConstraint.m_friction * totalImpulse);
181  frictionConstraint.m_upperLimit = frictionConstraint.m_friction * totalImpulse;
182  btScalar residual = resolveSingleConstraintRowGeneric(frictionConstraint);
183  leastSquaredResidual = btMax(leastSquaredResidual, residual * residual);
184 
185  if (frictionConstraint.m_multiBodyA)
186  frictionConstraint.m_multiBodyA->setPosUpdated(false);
187  if (frictionConstraint.m_multiBodyB)
188  frictionConstraint.m_multiBodyB->setPosUpdated(false);
189  }
190  }
191  }
192  }
193  return leastSquaredResidual;

228 btScalar btMultiBodyConstraintSolver::resolveSingleConstraintRowGeneric(const btMultiBodySolverConstraint& c)
229 {
230  btScalar deltaImpulse = c.m_rhs - btScalar(c.m_appliedImpulse) * c.m_cfm;
231  btScalar deltaVelADotn = 0;
232  btScalar deltaVelBDotn = 0;
233  btSolverBody* bodyA = 0;
234  btSolverBody* bodyB = 0;
235  int ndofA = 0;
236  int ndofB = 0;
237 
238  if (c.m_multiBodyA)
239  {
240  ndofA = c.m_multiBodyA->getNumDofs() + 6;
241  for (int i = 0; i < ndofA; ++i)
242  deltaVelADotn += m_data.m_jacobians[c.m_jacAindex + i] * m_data.m_deltaVelocities[c.m_deltaVelAindex + i];
243  }
244  else if (c.m_solverBodyIdA >= 0)
245  {
246  bodyA = &m_tmpSolverBodyPool[c.m_solverBodyIdA];
247  deltaVelADotn += c.m_contactNormal1.dot(bodyA->internalGetDeltaLinearVelocity()) + c.m_relpos1CrossNormal.dot(bodyA->internalGetDeltaAngularVelocity());
248  }
249 
250  if (c.m_multiBodyB)
251  {
252  ndofB = c.m_multiBodyB->getNumDofs() + 6;
253  for (int i = 0; i < ndofB; ++i)
254  deltaVelBDotn += m_data.m_jacobians[c.m_jacBindex + i] * m_data.m_deltaVelocities[c.m_deltaVelBindex + i];
255  }
256  else if (c.m_solverBodyIdB >= 0)
257  {
258  bodyB = &m_tmpSolverBodyPool[c.m_solverBodyIdB];
259  deltaVelBDotn += c.m_contactNormal2.dot(bodyB->internalGetDeltaLinearVelocity()) + c.m_relpos2CrossNormal.dot(bodyB->internalGetDeltaAngularVelocity());
260  }
261 
262  deltaImpulse -= deltaVelADotn * c.m_jacDiagABInv; //m_jacDiagABInv = 1./denom
263  deltaImpulse -= deltaVelBDotn * c.m_jacDiagABInv;
264  const btScalar sum = btScalar(c.m_appliedImpulse) + deltaImpulse;
265 
266  if (sum < c.m_lowerLimit)
267  {
268  deltaImpulse = c.m_lowerLimit - c.m_appliedImpulse;
269  c.m_appliedImpulse = c.m_lowerLimit;
270  }
271  else if (sum > c.m_upperLimit)
272  {
273  deltaImpulse = c.m_upperLimit - c.m_appliedImpulse;
274  c.m_appliedImpulse = c.m_upperLimit;
275  }
276  else
277  {
278  c.m_appliedImpulse = sum;
279  }
280 
281  if (c.m_multiBodyA)
282  {
283  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[c.m_jacAindex], deltaImpulse, c.m_deltaVelAindex, ndofA);
284 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
285  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
286  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
287  c.m_multiBodyA->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[c.m_jacAindex], deltaImpulse);
288 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
289  }
290  else if (c.m_solverBodyIdA >= 0)
291  {
292  bodyA->internalApplyImpulse(c.m_contactNormal1 * bodyA->internalGetInvMass(), c.m_angularComponentA, deltaImpulse);
293  }
294  if (c.m_multiBodyB)
295  {
296  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[c.m_jacBindex], deltaImpulse, c.m_deltaVelBindex, ndofB);
297 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
298  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
299  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
300  c.m_multiBodyB->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[c.m_jacBindex], deltaImpulse);
301 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
302  }
303  else if (c.m_solverBodyIdB >= 0)
304  {
305  bodyB->internalApplyImpulse(c.m_contactNormal2 * bodyB->internalGetInvMass(), c.m_angularComponentB, deltaImpulse);
306  }
307  btScalar deltaVel = deltaImpulse / c.m_jacDiagABInv;
308  return deltaVel;
309 }

311 btScalar btMultiBodyConstraintSolver::resolveConeFrictionConstraintRows(const btMultiBodySolverConstraint& cA1, const btMultiBodySolverConstraint& cB)
312 {
313  int ndofA = 0;
314  int ndofB = 0;
315  btSolverBody* bodyA = 0;
316  btSolverBody* bodyB = 0;
317  btScalar deltaImpulseB = 0.f;
318  btScalar sumB = 0.f;
319  {
320  deltaImpulseB = cB.m_rhs - btScalar(cB.m_appliedImpulse) * cB.m_cfm;
321  btScalar deltaVelADotn = 0;
322  btScalar deltaVelBDotn = 0;
323  if (cB.m_multiBodyA)
324  {
325  ndofA = cB.m_multiBodyA->getNumDofs() + 6;
326  for (int i = 0; i < ndofA; ++i)
327  deltaVelADotn += m_data.m_jacobians[cB.m_jacAindex + i] * m_data.m_deltaVelocities[cB.m_deltaVelAindex + i];
328  }
329  else if (cB.m_solverBodyIdA >= 0)
330  {
331  bodyA = &m_tmpSolverBodyPool[cB.m_solverBodyIdA];
332  deltaVelADotn += cB.m_contactNormal1.dot(bodyA->internalGetDeltaLinearVelocity()) + cB.m_relpos1CrossNormal.dot(bodyA->internalGetDeltaAngularVelocity());
333  }
334 
335  if (cB.m_multiBodyB)
336  {
337  ndofB = cB.m_multiBodyB->getNumDofs() + 6;
338  for (int i = 0; i < ndofB; ++i)
339  deltaVelBDotn += m_data.m_jacobians[cB.m_jacBindex + i] * m_data.m_deltaVelocities[cB.m_deltaVelBindex + i];
340  }
341  else if (cB.m_solverBodyIdB >= 0)
342  {
343  bodyB = &m_tmpSolverBodyPool[cB.m_solverBodyIdB];
344  deltaVelBDotn += cB.m_contactNormal2.dot(bodyB->internalGetDeltaLinearVelocity()) + cB.m_relpos2CrossNormal.dot(bodyB->internalGetDeltaAngularVelocity());
345  }
346 
347  deltaImpulseB -= deltaVelADotn * cB.m_jacDiagABInv; //m_jacDiagABInv = 1./denom
348  deltaImpulseB -= deltaVelBDotn * cB.m_jacDiagABInv;
349  sumB = btScalar(cB.m_appliedImpulse) + deltaImpulseB;
350  }
351 
352  btScalar deltaImpulseA = 0.f;
353  btScalar sumA = 0.f;
354  const btMultiBodySolverConstraint& cA = cA1;
355  {
356  {
357  deltaImpulseA = cA.m_rhs - btScalar(cA.m_appliedImpulse) * cA.m_cfm;
358  btScalar deltaVelADotn = 0;
359  btScalar deltaVelBDotn = 0;
360  if (cA.m_multiBodyA)
361  {
362  ndofA = cA.m_multiBodyA->getNumDofs() + 6;
363  for (int i = 0; i < ndofA; ++i)
364  deltaVelADotn += m_data.m_jacobians[cA.m_jacAindex + i] * m_data.m_deltaVelocities[cA.m_deltaVelAindex + i];
365  }
366  else if (cA.m_solverBodyIdA >= 0)
367  {
368  bodyA = &m_tmpSolverBodyPool[cA.m_solverBodyIdA];
369  deltaVelADotn += cA.m_contactNormal1.dot(bodyA->internalGetDeltaLinearVelocity()) + cA.m_relpos1CrossNormal.dot(bodyA->internalGetDeltaAngularVelocity());
370  }
371 
372  if (cA.m_multiBodyB)
373  {
374  ndofB = cA.m_multiBodyB->getNumDofs() + 6;
375  for (int i = 0; i < ndofB; ++i)
376  deltaVelBDotn += m_data.m_jacobians[cA.m_jacBindex + i] * m_data.m_deltaVelocities[cA.m_deltaVelBindex + i];
377  }
378  else if (cA.m_solverBodyIdB >= 0)
379  {
380  bodyB = &m_tmpSolverBodyPool[cA.m_solverBodyIdB];
381  deltaVelBDotn += cA.m_contactNormal2.dot(bodyB->internalGetDeltaLinearVelocity()) + cA.m_relpos2CrossNormal.dot(bodyB->internalGetDeltaAngularVelocity());
382  }
383 
384  deltaImpulseA -= deltaVelADotn * cA.m_jacDiagABInv; //m_jacDiagABInv = 1./denom
385  deltaImpulseA -= deltaVelBDotn * cA.m_jacDiagABInv;
386  sumA = btScalar(cA.m_appliedImpulse) + deltaImpulseA;
387  }
388  }
389 
390  if (sumA * sumA + sumB * sumB >= cA.m_lowerLimit * cB.m_lowerLimit)
391  {
392  btScalar angle = btAtan2(sumA, sumB);
393  btScalar sumAclipped = btFabs(cA.m_lowerLimit * btSin(angle));
394  btScalar sumBclipped = btFabs(cB.m_lowerLimit * btCos(angle));
395 
396  if (sumA < -sumAclipped)
397  {
398  deltaImpulseA = -sumAclipped - cA.m_appliedImpulse;
399  cA.m_appliedImpulse = -sumAclipped;
400  }
401  else if (sumA > sumAclipped)
402  {
403  deltaImpulseA = sumAclipped - cA.m_appliedImpulse;
404  cA.m_appliedImpulse = sumAclipped;
405  }
406  else
407  {
408  cA.m_appliedImpulse = sumA;
409  }
410 
411  if (sumB < -sumBclipped)
412  {
413  deltaImpulseB = -sumBclipped - cB.m_appliedImpulse;
414  cB.m_appliedImpulse = -sumBclipped;
415  }
416  else if (sumB > sumBclipped)
417  {
418  deltaImpulseB = sumBclipped - cB.m_appliedImpulse;
419  cB.m_appliedImpulse = sumBclipped;
420  }
421  else
422  {
423  cB.m_appliedImpulse = sumB;
424  }
425  //deltaImpulseA = sumAclipped-cA.m_appliedImpulse;
426  //cA.m_appliedImpulse = sumAclipped;
427  //deltaImpulseB = sumBclipped-cB.m_appliedImpulse;
428  //cB.m_appliedImpulse = sumBclipped;
429  }
430  else
431  {
432  cA.m_appliedImpulse = sumA;
433  cB.m_appliedImpulse = sumB;
434  }
435 
436  if (cA.m_multiBodyA)
437  {
438  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[cA.m_jacAindex], deltaImpulseA, cA.m_deltaVelAindex, ndofA);
439 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
440  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
441  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
442  cA.m_multiBodyA->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[cA.m_jacAindex], deltaImpulseA);
443 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
444  }
445  else if (cA.m_solverBodyIdA >= 0)
446  {
447  bodyA->internalApplyImpulse(cA.m_contactNormal1 * bodyA->internalGetInvMass(), cA.m_angularComponentA, deltaImpulseA);
448  }
449  if (cA.m_multiBodyB)
450  {
451  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[cA.m_jacBindex], deltaImpulseA, cA.m_deltaVelBindex, ndofB);
452 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
453  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
454  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
455  cA.m_multiBodyB->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[cA.m_jacBindex], deltaImpulseA);
456 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
457  }
458  else if (cA.m_solverBodyIdB >= 0)
459  {
460  bodyB->internalApplyImpulse(cA.m_contactNormal2 * bodyB->internalGetInvMass(), cA.m_angularComponentB, deltaImpulseA);
461  }
462 
463  if (cB.m_multiBodyA)
464  {
465  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[cB.m_jacAindex], deltaImpulseB, cB.m_deltaVelAindex, ndofA);
466 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
467  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
468  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
469  cB.m_multiBodyA->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[cB.m_jacAindex], deltaImpulseB);
470 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
471  }
472  else if (cB.m_solverBodyIdA >= 0)
473  {
474  bodyA->internalApplyImpulse(cB.m_contactNormal1 * bodyA->internalGetInvMass(), cB.m_angularComponentA, deltaImpulseB);
475  }
476  if (cB.m_multiBodyB)
477  {
478  applyDeltaVee(&m_data.m_deltaVelocitiesUnitImpulse[cB.m_jacBindex], deltaImpulseB, cB.m_deltaVelBindex, ndofB);
479 #ifdef DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
480  //note: update of the actual velocities (below) in the multibody does not have to happen now since m_deltaVelocities can be applied after all iterations
481  //it would make the multibody solver more like the regular one with m_deltaVelocities being equivalent to btSolverBody::m_deltaLinearVelocity/m_deltaAngularVelocity
482  cB.m_multiBodyB->applyDeltaVeeMultiDof2(&m_data.m_deltaVelocitiesUnitImpulse[cB.m_jacBindex], deltaImpulseB);
483 #endif //DIRECTLY_UPDATE_VELOCITY_DURING_SOLVER_ITERATIONS
484  }
485  else if (cB.m_solverBodyIdB >= 0)
486  {
487  bodyB->internalApplyImpulse(cB.m_contactNormal2 * bodyB->internalGetInvMass(), cB.m_angularComponentB, deltaImpulseB);
488  }
489 
490  btScalar deltaVel = deltaImpulseA / cA.m_jacDiagABInv + deltaImpulseB / cB.m_jacDiagABInv;
491  return deltaVel;
492 }

497  btScalar& relaxation,
498  bool isFriction, btScalar desiredVelocity, btScalar cfmSlip)
499 {

501  btVector3 rel_pos1;
502  btVector3 rel_pos2;

504  btMultiBody* multiBodyA = solverConstraint.m_multiBodyA;
505  btMultiBody* multiBodyB = solverConstraint.m_multiBodyB;
506 
507  const btVector3& pos1 = cp.getPositionWorldOnA();
508  const btVector3& pos2 = cp.getPositionWorldOnB();
509

511  btSolverBody* bodyB = multiBodyB ? 0 : &m_tmpSolverBodyPool[solverConstraint.m_solverBodyIdB];

518  if (bodyB)
519  rel_pos2 = pos2 - bodyB->getWorldTransform().getOrigin();
520 
521  relaxation = infoGlobal.m_sor;
522 
523  btScalar invTimeStep = btScalar(1) / infoGlobal.m_timeStep;
524 
525  //cfm = 1 / ( dt * kp + kd )
526  //erp = dt * kp / ( dt * kp + kd )
527 
528  btScalar cfm;
529  btScalar erp;
530  if (isFriction)
531  {
532  cfm = infoGlobal.m_frictionCFM;
533  erp = infoGlobal.m_frictionERP;
534  }
535  else
536  {
537  cfm = infoGlobal.m_globalCfm;
538  erp = infoGlobal.m_erp2;
539 
540  if ((cp.m_contactPointFlags & BT_CONTACT_FLAG_HAS_CONTACT_CFM) || (cp.m_contactPointFlags & BT_CONTACT_FLAG_HAS_CONTACT_ERP))
541  {
542  if (cp.m_contactPointFlags & BT_CONTACT_FLAG_HAS_CONTACT_CFM)
543  cfm = cp.m_contactCFM;
544  if (cp.m_contactPointFlags & BT_CONTACT_FLAG_HAS_CONTACT_ERP)
545  erp = cp.m_contactERP;
546  }
547  else
548  {
549  if (cp.m_contactPointFlags & BT_CONTACT_FLAG_CONTACT_STIFFNESS_DAMPING)
550  {
551  btScalar denom = (infoGlobal.m_timeStep * cp.m_combinedContactStiffness1 + cp.m_combinedContactDamping1);
552  if (denom < SIMD_EPSILON)
553  {
554  denom = SIMD_EPSILON;
555  }
556  cfm = btScalar(1) / denom;
557  erp = (infoGlobal.m_timeStep * cp.m_combinedContactStiffness1) / denom;
558  }
559  }
560  }
561 
562  cfm *= invTimeStep;
563 
564  if (multiBodyA)
565  {
566  if (solverConstraint.m_linkA < 0)
567  {
568  rel_pos1 = pos1 - multiBodyA->getBasePos();
569  }
570  else
571  {
572  rel_pos1 = pos1 - multiBodyA->getLink(solverConstraint.m_linkA).m_cachedWorldTransform.getOrigin();
573  }

575 
576  solverConstraint.m_deltaVelAindex = multiBodyA->getCompanionId();

578  if (solverConstraint.m_deltaVelAindex < 0)

582  m_data.m_deltaVelocities.resize(m_data.m_deltaVelocities.size() + ndofA);
583  }
584  else
585  {
586  btAssert(m_data.m_deltaVelocities.size() >= solverConstraint.m_deltaVelAindex + ndofA);
587  }
588 
589  solverConstraint.m_jacAindex = m_data.m_jacobians.size();
590  m_data.m_jacobians.resize(m_data.m_jacobians.size() + ndofA);
591  m_data.m_deltaVelocitiesUnitImpulse.resize(m_data.m_deltaVelocitiesUnitImpulse.size() + ndofA);
592  btAssert(m_data.m_jacobians.size() == m_data.m_deltaVelocitiesUnitImpulse.size());
593 
594  btScalar* jac1 = &m_data.m_jacobians[solverConstraint.m_jacAindex];
595  multiBodyA->fillContactJacobianMultiDof(solverConstraint.m_linkA, cp.getPositionWorldOnA(), contactNormal, jac1, m_data.scratch_r, m_data.scratch_v, m_data.scratch_m);
596  btScalar* delta = &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacAindex];
597  multiBodyA->calcAccelerationDeltasMultiDof(&m_data.m_jacobians[solverConstraint.m_jacAindex], delta, m_data.scratch_r, m_data.scratch_v);
598 
599  btVector3 torqueAxis0 = rel_pos1.cross(contactNormal);
600  solverConstraint.m_relpos1CrossNormal = torqueAxis0;
601  solverConstraint.m_contactNormal1 = contactNormal;
602  }
603  else
604  {
605  btVector3 torqueAxis0 = rel_pos1.cross(contactNormal);
606  solverConstraint.m_relpos1CrossNormal = torqueAxis0;
607  solverConstraint.m_contactNormal1 = contactNormal;
608  solverConstraint.m_angularComponentA = rb0 ? rb0->getInvInertiaTensorWorld() * torqueAxis0 * rb0->getAngularFactor() : btVector3(0, 0, 0);
609  }
610 
611  if (multiBodyB)
612  {
613  if (solverConstraint.m_linkB < 0)
614  {
615  rel_pos2 = pos2 - multiBodyB->getBasePos();
616  }
617  else
618  {
619  rel_pos2 = pos2 - multiBodyB->getLink(solverConstraint.m_linkB).m_cachedWorldTransform.getOrigin();
620  }
621 
622  const int ndofB = multiBodyB->getNumDofs() + 6;
623 
624  solverConstraint.m_deltaVelBindex = multiBodyB->getCompanionId();
625  if (solverConstraint.m_deltaVelBindex < 0)
626  {
627  solverConstraint.m_deltaVelBindex = m_data.m_deltaVelocities.size();
628  multiBodyB->setCompanionId(solverConstraint.m_deltaVelBindex);
629  m_data.m_deltaVelocities.resize(m_data.m_deltaVelocities.size() + ndofB);
630  }
631 
632  solverConstraint.m_jacBindex = m_data.m_jacobians.size();
633 
634  m_data.m_jacobians.resize(m_data.m_jacobians.size() + ndofB);
635  m_data.m_deltaVelocitiesUnitImpulse.resize(m_data.m_deltaVelocitiesUnitImpulse.size() + ndofB);
636  btAssert(m_data.m_jacobians.size() == m_data.m_deltaVelocitiesUnitImpulse.size());
637 
638  multiBodyB->fillContactJacobianMultiDof(solverConstraint.m_linkB, cp.getPositionWorldOnB(), -contactNormal, &m_data.m_jacobians[solverConstraint.m_jacBindex], m_data.scratch_r, m_data.scratch_v, m_data.scratch_m);
639  multiBodyB->calcAccelerationDeltasMultiDof(&m_data.m_jacobians[solverConstraint.m_jacBindex], &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacBindex], m_data.scratch_r, m_data.scratch_v);
640 
641  btVector3 torqueAxis1 = rel_pos2.cross(contactNormal);
642  solverConstraint.m_relpos2CrossNormal = -torqueAxis1;
643  solverConstraint.m_contactNormal2 = -contactNormal;
644  }
645  else
646  {
647  btVector3 torqueAxis1 = rel_pos2.cross(contactNormal);
648  solverConstraint.m_relpos2CrossNormal = -torqueAxis1;
649  solverConstraint.m_contactNormal2 = -contactNormal;
650 
651  solverConstraint.m_angularComponentB = rb1 ? rb1->getInvInertiaTensorWorld() * -torqueAxis1 * rb1->getAngularFactor() : btVector3(0, 0, 0);
652  }
653 
654  {
655  btVector3 vec;
656  btScalar denom0 = 0.f;
657  btScalar denom1 = 0.f;
658  btScalar* jacB = 0;
659  btScalar* jacA = 0;
660  btScalar* lambdaA = 0;
661  btScalar* lambdaB = 0;
662  int ndofA = 0;
663  if (multiBodyA)
664  {
665  ndofA = multiBodyA->getNumDofs() + 6;
666  jacA = &m_data.m_jacobians[solverConstraint.m_jacAindex];
667  lambdaA = &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacAindex];
668  for (int i = 0; i < ndofA; ++i)
669  {
670  btScalar j = jacA[i];
671  btScalar l = lambdaA[i];
672  denom0 += j * l;
673  }
674  }
675  else
676  {
677  if (rb0)
678  {
679  vec = (solverConstraint.m_angularComponentA).cross(rel_pos1);
680  denom0 = rb0->getInvMass() + contactNormal.dot(vec);
681  }
682  }
683  if (multiBodyB)
684  {
685  const int ndofB = multiBodyB->getNumDofs() + 6;
686  jacB = &m_data.m_jacobians[solverConstraint.m_jacBindex];
687  lambdaB = &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacBindex];
688  for (int i = 0; i < ndofB; ++i)
689  {
690  btScalar j = jacB[i];
691  btScalar l = lambdaB[i];
692  denom1 += j * l;
693  }
694  }
695  else
696  {
697  if (rb1)
698  {
699  vec = (-solverConstraint.m_angularComponentB).cross(rel_pos2);
700  denom1 = rb1->getInvMass() + contactNormal.dot(vec);
701  }
702  }
703 
704  btScalar d = denom0 + denom1 + cfm;
705  if (d > SIMD_EPSILON)
706  {
707  solverConstraint.m_jacDiagABInv = relaxation / (d);
708  }
709  else
710  {
711  //disable the constraint row to handle singularity/redundant constraint
712  solverConstraint.m_jacDiagABInv = 0.f;
713  }
714  }
715 
716  //compute rhs and remaining solverConstraint fields
717 
718  btScalar restitution = 0.f;
719  btScalar distance = 0;
720  if (!isFriction)
721  {
722  distance = cp.getDistance() + infoGlobal.m_linearSlop;
723  }
724  else
725  {
726  if (cp.m_contactPointFlags & BT_CONTACT_FLAG_FRICTION_ANCHOR)
727  {
728  distance = (cp.getPositionWorldOnA() - cp.getPositionWorldOnB()).dot(contactNormal);
729  }
730  }
731 
732  btScalar rel_vel = 0.f;
733  int ndofA = 0;
734  int ndofB = 0;
735  {
736  btVector3 vel1, vel2;
737  if (multiBodyA)
738  {
739  ndofA = multiBodyA->getNumDofs() + 6;
740  btScalar* jacA = &m_data.m_jacobians[solverConstraint.m_jacAindex];
741  for (int i = 0; i < ndofA; ++i)
742  rel_vel += multiBodyA->getVelocityVector()[i] * jacA[i];
743  }
744  else
745  {
746  if (rb0)
747  {
748  rel_vel += (rb0->getVelocityInLocalPoint(rel_pos1) +
749  (rb0->getTotalTorque() * rb0->getInvInertiaTensorWorld() * infoGlobal.m_timeStep).cross(rel_pos1) +
750  rb0->getTotalForce() * rb0->getInvMass() * infoGlobal.m_timeStep)
751  .dot(solverConstraint.m_contactNormal1);
752  }
753  }
754  if (multiBodyB)
755  {
756  ndofB = multiBodyB->getNumDofs() + 6;
757  btScalar* jacB = &m_data.m_jacobians[solverConstraint.m_jacBindex];
758  for (int i = 0; i < ndofB; ++i)
759  rel_vel += multiBodyB->getVelocityVector()[i] * jacB[i];
760  }
761  else
762  {
763  if (rb1)
764  {
765  rel_vel += (rb1->getVelocityInLocalPoint(rel_pos2) +
766  (rb1->getTotalTorque() * rb1->getInvInertiaTensorWorld() * infoGlobal.m_timeStep).cross(rel_pos2) +
767  rb1->getTotalForce() * rb1->getInvMass() * infoGlobal.m_timeStep)
768  .dot(solverConstraint.m_contactNormal2);
769  }
770  }
771 
772  solverConstraint.m_friction = cp.m_combinedFriction;
773 
774  if (!isFriction)
775  {
776  restitution = restitutionCurve(rel_vel, cp.m_combinedRestitution, infoGlobal.m_restitutionVelocityThreshold);
777  if (restitution <= btScalar(0.))
778  {
779  restitution = 0.f;
780  }
781  }
782  }
783 
785  //disable warmstarting for btMultiBody, it has issues gaining energy (==explosion)
786  if (0) //infoGlobal.m_solverMode & SOLVER_USE_WARMSTARTING)
787  {
788  solverConstraint.m_appliedImpulse = isFriction ? 0 : cp.m_appliedImpulse * infoGlobal.m_warmstartingFactor;
789 
790  if (solverConstraint.m_appliedImpulse)
791  {
792  if (multiBodyA)
793  {
794  btScalar impulse = solverConstraint.m_appliedImpulse;
795  btScalar* deltaV = &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacAindex];
796  multiBodyA->applyDeltaVeeMultiDof(deltaV, impulse);
797 
798  applyDeltaVee(deltaV, impulse, solverConstraint.m_deltaVelAindex, ndofA);
799  }
800  else
801  {
802  if (rb0)
803  bodyA->internalApplyImpulse(solverConstraint.m_contactNormal1 * bodyA->internalGetInvMass() * rb0->getLinearFactor(), solverConstraint.m_angularComponentA, solverConstraint.m_appliedImpulse);
804  }
805  if (multiBodyB)
806  {
807  btScalar impulse = solverConstraint.m_appliedImpulse;
808  btScalar* deltaV = &m_data.m_deltaVelocitiesUnitImpulse[solverConstraint.m_jacBindex];
809  multiBodyB->applyDeltaVeeMultiDof(deltaV, impulse);
810  applyDeltaVee(deltaV, impulse, solverConstraint.m_deltaVelBindex, ndofB);
811  }
812  else
813  {
814  if (rb1)
815  bodyB->internalApplyImpulse(-solverConstraint.m_contactNormal2 * bodyB->internalGetInvMass() * rb1->getLinearFactor(), -solverConstraint.m_angularComponentB, -(btScalar)solverConstraint.m_appliedImpulse);
816  }
817  }
818  }
819  else
820  {
821  solverConstraint.m_appliedImpulse = 0.f;
822  }
823 
824  solverConstraint.m_appliedPushImpulse = 0.f;
825 
826  {
827  btScalar positionalError = 0.f;
828  btScalar velocityError = restitution - rel_vel; // * damping; //note for friction restitution is always set to 0 (check above) so it is acutally velocityError = -rel_vel for friction
829  if (isFriction)
830  {
831  positionalError = -distance * erp / infoGlobal.m_timeStep;
832  }
833  else
834  {
835  if (distance > 0)
836  {
837  positionalError = 0;
838  velocityError -= distance / infoGlobal.m_timeStep;
839  }
840  else
841  {
842  positionalError = -distance * erp / infoGlobal.m_timeStep;
843  }
844  }
845 
846  btScalar penetrationImpulse = positionalError * solverConstraint.m_jacDiagABInv;
847  btScalar velocityImpulse = velocityError * solverConstraint.m_jacDiagABInv;
848 
849  if (!isFriction)
850  {
851  // if (!infoGlobal.m_splitImpulse || (penetration > infoGlobal.m_splitImpulsePenetrationThreshold))
852  {
853  //combine position and velocity into rhs
854  solverConstraint.m_rhs = penetrationImpulse + velocityImpulse;
855  solverConstraint.m_rhsPenetration = 0.f;
856  }
857  /*else
858  {
859  //split position and velocity into rhs and m_rhsPenetration
860  solverConstraint.m_rhs = velocityImpulse;
861  solverConstraint.m_rhsPenetration = penetrationImpulse;
862  }
863  */
864  solverConstraint.m_lowerLimit = 0;
865  solverConstraint.m_upperLimit = 1e10f;
866  }
867  else
868  {
869  solverConstraint.m_rhs = penetrationImpulse + velocityImpulse;
870  solverConstraint.m_rhsPenetration = 0.f;
871  solverConstraint.m_lowerLimit = -solverConstraint.m_friction;
872  solverConstraint.m_upperLimit = solverConstraint.m_friction;
873  }
874 
875  solverConstraint.m_cfm = cfm * solverConstraint.m_jacDiagABInv;
876  }
877 }
