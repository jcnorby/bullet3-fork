546 void btMultiBody::compTreeLinkVelocities(btVector3 *omega, btVector3 *vel) const
547 {
548  int num_links = getNumLinks();
549  // Calculates the velocities of each link (and the base) in its local frame
550  const btQuaternion& base_rot = getWorldToBaseRot();
551  omega[0] = quatRotate(base_rot, getBaseOmega());
552  vel[0] = quatRotate(base_rot, getBaseVel());
553 
554  for (int i = 0; i < num_links; ++i)
555  {
556  const btMultibodyLink& link = getLink(i);
557  const int parent = link.m_parent;
558 
559  // transform parent vel into this frame, store in omega[i+1], vel[i+1]
560  spatialTransform(btMatrix3x3(link.m_cachedRotParentToThis), link.m_cachedRVector,
561  omega[parent + 1], vel[parent + 1],
562  omega[i + 1], vel[i + 1]);
563 
564  // now add qidot * shat_i
565  const btScalar* jointVel = getJointVelMultiDof(i);
566  for (int dof = 0; dof < link.m_dofCount; ++dof)
567  {
568  omega[i + 1] += jointVel[dof] * link.getAxisTop(dof);
569  vel[i + 1] += jointVel[dof] * link.getAxisBottom(dof);
570  }
571  }
